	(\text{Given series})
	 & = \sum_{n = 0}^\infty \frac{1}{{(2n + 1)}^2}                                 \\
	 & = \sum_{n = 1}^\infty \frac{1}{n^2} - \sum_{n = 1}^\infty \frac{1}{{(2n)}^2} \\
