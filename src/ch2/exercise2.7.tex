\begin{align*}
	(\text{Given series})
	&= \sum_{n = 1}^\infty \frac{{(-1)}^{n + 1}}{n^2} \\
	&= \sum_{n = 1}^\infty \frac{1}{n^2} - 2 \sum_{n = 1}^\infty \frac{1}{{(2n)}^2} \\
	&= \half \sum_{n = 0}^\infty \frac{1}{n^2} \\
	&= \half \cdot \frac{\pi^2}{6} \\
	&= \frac{\pi^2}{12}
