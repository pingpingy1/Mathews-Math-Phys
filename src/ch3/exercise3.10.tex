We shall perform a contour integral of $f(z) := \frac{1}{1 + z^4}$ along \reffig{}~\ref{fig:problem3-10}.
\[
	\oint_C dz f(z)
	= 2i\pi \res{z = e^{i\quarterpi}} f(z)
	= 2i\pi \cdot \evalAt{\frac{1}{4x^3}}{z = e^{i\quarterpi}}
	= \halfpi e^{-i\quarterpi}
\]
We also have
\begin{align*}
	\oint_C dx f(z)
	 & = \int_{0}^{R} \frac{dx}{1 + x^4}
	+ \int_0^{\halfpi} d\theta \frac{iRe^{i\theta}}{1 + R^4 e^{4i\theta}}
	+ \int_{R}^0 dy \frac{i}{1 + {(iy)}^4}                                              \\
	 & = (1 - i) \int_{0}^{R} \frac{dx}{1 + x^4} + O\left( R^{-3} \right)               \\
	 & \xrightarrow{R \rightarrow \infty} (1 - i) \int_{0}^{\infty} \frac{dx}{1 + x^4}.
\end{align*}
\[
	\therefore \int_{0}^{\infty} \frac{dx}{1 + x^4}
	= \frac{1}{1 - i} \cdot \halfpi e^{-i\quarterpi}
	= \frac{\pi}{2\sqrt{2}}
\]
